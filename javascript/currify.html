柯里化：把接收多个参数的函数变换成接受一个单一参数的函数，并且返回接受余下参数而且返回结果的新函数。

Function.length 可以获取函数期望的参数长度。

柯里化的实现：

function curry(fn, args = []){
	return function(){
		let params = [...args, ...arguments]
		//check if already collected enough params
		if(params.length < fn.length){
			return curry(fn, params)
		}else{
			return fn(...params);
		}
	}
}

let sum = function(a,b,c,d){
	return a+b+c+d;
}

let sumFn = curry(sum)

console.log(sumFn(1,2,3)(4)) //10
console.log(sumFn(1,2)(3)(4)) //10
console.log(sumFn(1)(2)(3)(4)) //10
