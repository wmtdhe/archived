# arguments是数组吗，如何转化为数组

因为本身arguments不能调用数组方法，它是一个另外一种对象类型，只不过属性从0开始排，依次为0, 1, 2...最后还有callee和length属性。我们把这样的对象称为类数组。

常见的类数组还有：

  • getElementsByTagName/ClassName() -- 获得的HTMLCollection
  • querySelectorAll获得的nodeList

那很多数组的方法就不能用了，如何把他们转换为数组呢？

# 1. Array.prototype.slice.call()

function sum(a,b){
    let args = Array.prototype.slice.call(arguments);
    console.log(args.reduce((sum,cur)=>sum+cur)) // args可以调用数组原生方法
}
sum(1,2) // 3

# 2. Array.from()

function sum(a,b){
    let args = Array.from(arguments);
    console.log(args.reduce((sum,cur)=>sum+cur)) // args可以调用数组原生方法
}
sum(1,2) // 3

# 3. ES6展开运算符

function sum(a,b){
    //arguments is iterable
    let args = [...arguments];
    console.log(args.reduce((sum,cur)=>sum+cur)) // args可以调用数组原生方法
}
sum(1,2) // 3

# 4.利用concat+apply

function sum(a,b){
    let args = Array.prototype.concat.apply([],arguments);//apply方法会把第二个参数展开
    console.log(args.reduce((sum,cur)=>sum+cur)) // args可以调用数组原生方法
}
sum(1,2) //3

# forEach中的return有效果吗，如何中断forEach循环

在forEach中用return不会返回，函数会继续执行。

let nums = [1,2,3];
nums.forEach((v,i)=>{
    return ; // 无效
})

中断方法：

  1. 用try监视代码，在需要中断的地方抛出异常。
  2. 用every和some替代forEach。当every返回false时中断，some返回true时中断。

# 数组扁平化

# 1. ES6的flat方法

arr = arr.flat(Infinity);

# 2. reduce+concat迭代

function flatten(arr){
    return arr.reduce((pre,cur)=>{
        return pre.concat(Array.isArray(cur)?flatten(cur):cur);
    },[]) 
}
//with depth
function flatten(arr,depth){
    if(depth<=0){
        return arr;
    }
    depth--;
    return arr.reduce((pre,cur)=>{
       return pre.concat(Array.isArray(cur)?flatten(cur,depth):cur) 
    },[])
}

# 3.扩展运算符

while(arr.some(Array.isArray)){
    arr = [].concat(...arr);
    // [].concat(1,2,[4,5]) --> [1,2,4,5]
}

# 4.普通递归

let result = []
function flatten(arr){
    for(let i=0;i<arr.length;i++){
        if(Array.isArray(arr[i])){
            flatten(arr[i])
        }else{
            result.push(arr[i])
        }
    }
}
