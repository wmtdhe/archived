# 原型链

# 原型对象和构造函数的关系

在JavaScript中，每定义一个函数数据类型（普通函数，类）的时候，都会天使自带一个prototype属性，这个属性指向函数的原型对象。

当函数经过new调用时，这个函数就成就了构造函数，返回一个全新的实例对象，这个实例对象有一个__proto__属性，指向构造函数的原型对象(prototype)。

# 原型链

JavaScript对象通过prototype指向父类对象，直到指向Object对象为止，这就形成了一个原型指向的链条。

  • 对象的hasOwnProperty()可以用来检查对象自身中是否含有该属性
  • 使用in检查对象中是否含有某个属性时，如果对象中没有但是原型链中有，会返回true。
let obj = {
    value: 3
}
console.log('value' in obj) // true

# JS如何实现继承

# 第一种：借助call

function Parent(){
    this.name = 'name';
}

function Child(){
    Parent.call(this)
}

这样写的子类虽然能够拿到父类的属性，但是父类原型对象中一旦存在方法，子类无法继承。

# 第二种：借助原型链

function Parent(){
    this.name = 'name';
    this.play = [1,2,3]
}

function Child(){
    this.type = 'child'
}
Child.prototype = new Parent();

然而这种方法实例间共用同一个原型对象

var s1 = new Child()
var s2 = new Child()
s1.play.push(4)
console.log(s1.play,s2.play)
// (4)[1, 2, 3, 4] (4)[1, 2, 3, 4]

# 第三种：将前两种结合：

function Parent(){
    this.name = 'name';
    this.play = [1,2,3];
}

function Child(){
    Person.call(this);
    this.type = 'type'
}

Child.prototype = new Parent();
var s1 = new Child();
var s2 = new Child();
s1.play.push(4);
console.log(s1.play, s2.play)
// (4)[1, 2, 3, 4] (3)[1, 2, 3]

之前的问题都得以解决。但这里有新增一个问题，那就是Parent的构造函数会多执行一次Child.prototype = new Parent()。以下为优化

# 第四种：组合继承的优化1

function Parent(){
    this.name = 'name';
    this.play = [1,2,3]
}
function Child(){
    Parent.call(this);
    this.type = 'type'
}
Child.prototype = Parent.prototype;

这里直接将父类原型对象给到子类，父类构造函数只执行了一次，而且父类属性和方法均能访问，但是：

var s1 = new Child();
var s2 = new Child();
console.log(s1)

子类实例的constructor是Parent而不是Child。

# 第五种（推荐）：组合继承的优化2

function Parent(){
    this.name = 'name';
    this.play = [1,2,3]
}
function Child(){
    Parent.call(this);
    this.type = 'type';
}
Child.prototype = Object.create(Parent.prototype);
Child.prototype.constructor = Child;

这叫寄生组合继承

# ES6的extends被编译后JS代码

babel将ES6代码编译为ES5，让一些不支持新语法的浏览器也能运行。

function _possibleConstructorReturn(self, call) {
    // ...
    return call && (typeof call === 'object' || typeof call === 'function') ? call : self;
}

function _inherits(subClass, superClass) {
    // ...
    // --------- 寄生组合继承
    subClass.prototype = Object.create(superClass && superClass.prototype, {
        constructor: {
            value: subClass,
            enumerable: false,
            writable: true,
            configurable: true
        }
    });
    if (superClass) Object.setPrototypeOf ? Object.setPrototypeOf(subClass, superClass) : subClass.__proto__ = superClass;
}


var Parent = function Parent() {
    // 验证是否是 Parent 构造出来的 this
    _classCallCheck(this, Parent);
};

var Child = (function (_Parent) {
    _inherits(Child, _Parent);

    function Child() {
        _classCallCheck(this, Child);

        return _possibleConstructorReturn(this, (Child.__proto__ || Object.getPrototypeOf(Child)).apply(this, arguments));
    }

    return Child;
}(Parent));
